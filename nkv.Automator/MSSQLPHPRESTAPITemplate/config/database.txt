<?php
class Database{
    private $host;
    private $db_name;
    private $username;
    private $password;
	private $port;
    public $conn;

    function __construct($dbName="{dbName}") {
        $this->db_name = $dbName;
        $this->host = "{hostName}";
        $this->username = "{userName}";
        $this->password = "{password}";
        $this->port = "{port}";
        $this->conn = null;
    }
    public function getConnection(){
        try
        {   
		    if($this->port){
                $Server = $this->host.','.$this->port;
                $this->conn = new PDO("sqlsrv:Server=" . $Server . ";Database=" . $this->db_name, $this->username, $this->password);
                $this->conn->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
                return $this->conn;
            }
            else{
			    $this->conn = new PDO("sqlsrv:Server=" . $this->host . ";Database=" . $this->db_name, $this->username, $this->password);
                $this->conn->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
                return $this->conn;
		    }
        }catch(PDOException $exception){
            echo "Connection error: " . $exception->getMessage();
            return null;
        }
 
        
    }
}
?>
